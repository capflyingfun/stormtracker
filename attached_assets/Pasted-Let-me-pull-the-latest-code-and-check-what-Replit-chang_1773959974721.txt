Replace buildNexradFrames() with this version that snaps times to exact 5-minute UTC boundaries with zero seconds, and skips the most recent slot to avoid the processing lag:

async function buildNexradFrames(){
  const frames=[];
  const now=new Date();
  // Snap to the PREVIOUS 5-minute boundary (subtract 10 min to avoid processing lag)
  const base=new Date(now);
  base.setUTCSeconds(0,0);
  base.setUTCMinutes(Math.floor(base.getUTCMinutes()/5)*5);
  // Go back 10 minutes from snapped time to ensure mosaic exists
  const safeBase=new Date(base.getTime() - 10*60000);
  
  for(let i=24;i>=0;i--){
    const t=new Date(safeBase.getTime() - i*5*60000);
    // Build clean ISO time with exactly zero seconds: YYYY-MM-DDTHH:MM:00Z
    const isoTime=t.getUTCFullYear()+'-'+
      String(t.getUTCMonth()+1).padStart(2,'0')+'-'+
      String(t.getUTCDate()).padStart(2,'0')+'T'+
      String(t.getUTCHours()).padStart(2,'0')+':'+
      String(t.getUTCMinutes()).padStart(2,'0')+':00Z';
    frames.push({time:Math.floor(t.getTime()/1000), type:'past', isoTime, wms:true});
  }
  return frames;
}
