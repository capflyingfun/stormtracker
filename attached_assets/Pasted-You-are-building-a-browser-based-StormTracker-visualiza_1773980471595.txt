Generating polygons:

function makePolygon(binPoints) {
  if (binPoints.length < 3) return null;
  const pts = turf.points(binPoints.map(p => [p.lon, p.lat]));
  let poly = turf.concave(pts, { maxEdge: 1 });
  if (!poly) poly = turf.convex(pts);
  return poly;
}
--------------------------------

function checkUserLocation(userLat, userLon, polygons) {
  const pt = turf.point([userLon, userLat]);
  for (const level in polygons) {
    const poly = polygons[level];
    if (poly && turf.booleanPointInPolygon(pt, poly)) {
      alert("You are inside " + level + " dBZ zone");
    }
  }
}
