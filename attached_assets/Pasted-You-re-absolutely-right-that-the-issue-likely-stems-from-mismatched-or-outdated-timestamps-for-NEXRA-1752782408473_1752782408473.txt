🧩 Step 2: Full JavaScript Code Snippet

<script>
// Approximate user location (replace with Geolocation API if needed)
const userLat = 30.4;
const userLng = -87.2;

// Common NEXRAD radar stations with lat/lon
const radarSites = {
  KTLX: { lat: 35.33, lon: -97.28 },
  KEVX: { lat: 30.57, lon: -85.93 },
  KMOB: { lat: 30.68, lon: -88.24 },
  KLIX: { lat: 30.34, lon: -89.82 },
  KJGX: { lat: 32.67, lon: -83.35 },
  // Add more stations as needed
};

// Get closest radar site
function getNearestRadar(lat, lon) {
  let nearest = null;
  let minDist = Infinity;
  for (const [code, pos] of Object.entries(radarSites)) {
    const dist = Math.hypot(lat - pos.lat, lon - pos.lon);
    if (dist < minDist) {
      minDist = dist;
      nearest = code;
    }
  }
  return nearest;
}

const station = getNearestRadar(userLat, userLng);
let radarLayer;
let currentFrame = 0;
let tileLayers = [];

function fetchTimestampsAndAnimate() {
  fetch(`https://mesonet.agron.iastate.edu/json/radar.py?operation=list&radar=${station}&product=ridge`)
    .then(res => res.json())
    .then(data => {
      const timestamps = data.scans.slice(-8);
      tileLayers = timestamps.map(ts =>
        L.tileLayer(`https://mesonet.agron.iastate.edu/cache/tile.py/1.0.0/ridge::${station}::${ts}/{z}/{x}/{y}.png`, {
          tileSize: 256,
          opacity: 0.65,
          zIndex: 10
        })
      );
      animateRadar();
    });
}

function showFrame(index) {
  if (radarLayer) map.removeLayer(radarLayer);
  radarLayer = tileLayers[index];
  radarLayer.addTo(map);
}

function animateRadar() {
  showFrame(currentFrame);
  currentFrame = (currentFrame + 1) % tileLayers.length;
  setTimeout(animateRadar, 500);
}

// Start it
fetchTimestampsAndAnimate();
</script>
